#-------------------------------------------------------------------------------------------
# CMake info
#-------------------------------------------------------------------------------------------
cmake_minimum_required(VERSION 3.10)

#-------------------------------------------------------------------------------------------
# Graphic APIs
#-------------------------------------------------------------------------------------------

#-------------------------------------------------------------------------------------------
# DirectX
#-------------------------------------------------------------------------------------------
if ("${HYPERENGINE_ARCH}" STREQUAL "Windows")
    add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/DirectX11)
    add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/DirectX12)
endif ()

#-------------------------------------------------------------------------------------------
# OpenGL
#-------------------------------------------------------------------------------------------
find_package(OpenGL)
if (OPENGL_FOUND)
    add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/OpenGL33)
    add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/OpenGL46)
endif ()

#-------------------------------------------------------------------------------------------
# Vulkan
#-------------------------------------------------------------------------------------------
find_package(Vulkan)
if (Vulkan_FOUND)
    add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/Vulkan)
endif ()
